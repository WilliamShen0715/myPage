int main(int argc, char*argv[])
{
    FILE *ifp, *ofp;
    int ftwo[32],dtwo[64];
    char fbit[32],dbit[64];
    int i,i2,i3,i4,i5,i6;
    float f;
    double d;
    ifp = fopen(argv[1],"r");
    ofp = fopen("output_a","w");
    fscanf(ifp,"%f",&f);
    fscanf(ifp,"%lf",&d);
    fscanf(ifp,"%s",fbit);
    fscanf(ifp,"%s",dbit);
    int* fipt;
//f to bit
    fipt = (int*)&f;
    for(i=sizeof(f)*8-1;i>=0;i--){
        ftwo[i] = *fipt&1;
        *fipt>>=1;
    }
    for(i2=0;i2<(int)sizeof(f)*8;i2++){
        fprintf(ofp,"%d",ftwo[i2]);
    }
    putc('\n',ofp);
//d to bit
    long long int* dipt;
    dipt = (long long int*)&d;
    for(i3=sizeof(d)*8-1;i3>=0;i3--){
        dtwo[i3] = *dipt&1;
        *dipt>>=1;
    }
    for(i4=0;i4<(int)sizeof(d)*8;i4++){
        fprintf(ofp,"%d",dtwo[i4]);
    }
    putc('\n',ofp);
//f bit to f
    *fipt = 0;
    for(i5=0;i5<(int)sizeof(f)*8-1;i5++){

        *fipt += fbit[i5]-'0';
        *fipt<<=1;
        //printf("%d\n",*fipt);
    }
    fprintf(ofp,"%.100g",f);
    putc('\n',ofp);
//d bit to d
    *dipt = 0;
    for(i6=0;i6<(int)sizeof(d)*8-1;i6++){
        *dipt |= dbit[i6]-'0';
        *dipt<<=1;
    }
    fprintf(ofp,"%.100g",d);
    putc('\n',ofp);
}
